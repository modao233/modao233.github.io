All, except last, words must be _complete_. e = "(^|.*[^\\pL])" + this.value.trim().split(/\s+/).join("([^\\pL]|[^\\pL].*[^\\pL])") + ".*$", n = RegExp(e, "i"); listItems.forEach(function(item) { item.removeAttribute('hidden'); }); listItems.filter(function(item) { i = item.querySelector('td').textContent.replace(/\s+/g, " "); return !n.test(i); }).forEach(function(item) { item.hidden = true; }); });
