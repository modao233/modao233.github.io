frp内网穿透

2020-09-23
:: tags:  #Software

server：

Frps.ini:

[common]
# 服务器端Frp监听的端口
bind_port = 7000
# http访问端口
vhost_http_port = 7002

./frps -c frps.ini

client:

Frpc.ini:

[common]
# 公网服务器的公网IP
server_addr = 106.13.182.38
# 服务器端Frp监听的端口
server_port = 7000

[web]
type = http
# 本地开放端口
local_port = 80
# 公网服务器的IP或者已解析的域名
custom_domains = 106.13.182.38

./frpc -c frpc.ini
